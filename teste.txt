git init //verificar projeto git

git status //verificar alteração

git config --list

git add "nome-do-arquivo" //adicionar repositório

git commit -m "mensagem" //empacotar

git push origin master //subir

git pull origin master //baixar

git clone //clonar

git branch //mostra a branch atual

git checkout "nome-da-branch" //trocar de branch

git checkout -b "nome-da-branch" //cria uma branch
